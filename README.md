# JpaSpringBootDemo
Practica con Jpa de Spring Boot.
* OneToOne
* OneToMany
* ManyToOne
* ManyToMany

**Tecnologias:**
* Spring Boot 3.3.2
* Java 17
* PostgreSQL 16

**Link Documentacion (Swagger):** http://localhost:8080/orm/swagger-ui/index.html

Creacion de tablas SQL:

```
CREATE DATABASE football;

CREATE TABLE IF NOT EXISTS public.competition
(
    id_competition integer NOT NULL GENERATED BY DEFAULT AS IDENTITY ( INCREMENT 1 START 1 MINVALUE 1 MAXVALUE 2147483647 CACHE 1 ),
    date_competition timestamp(6) without time zone NOT NULL,
    stadium character varying(255) COLLATE pg_catalog."default" NOT NULL,
    CONSTRAINT competition_pkey PRIMARY KEY (id_competition)
);

CREATE TABLE IF NOT EXISTS public.couch
(
    id_couch integer NOT NULL GENERATED BY DEFAULT AS IDENTITY ( INCREMENT 1 START 1 MINVALUE 1 MAXVALUE 2147483647 CACHE 1 ),
    status integer,
    name character varying(255) COLLATE pg_catalog."default",
    CONSTRAINT couch_pkey PRIMARY KEY (id_couch)
);

CREATE TABLE IF NOT EXISTS public.competition
(
    id_competition integer NOT NULL GENERATED BY DEFAULT AS IDENTITY ( INCREMENT 1 START 1 MINVALUE 1 MAXVALUE 2147483647 CACHE 1 ),
    date_competition timestamp(6) without time zone NOT NULL,
    stadium character varying(255) COLLATE pg_catalog."default" NOT NULL,
    CONSTRAINT competition_pkey PRIMARY KEY (id_competition)
);

CREATE TABLE IF NOT EXISTS public.club
(
    id_club integer NOT NULL GENERATED BY DEFAULT AS IDENTITY ( INCREMENT 1 START 1 MINVALUE 1 MAXVALUE 2147483647 CACHE 1 ),
    name character varying(255) COLLATE pg_catalog."default",
    status integer,
    id_couch integer NOT NULL,
    CONSTRAINT club_pkey PRIMARY KEY (id_club),
    CONSTRAINT uq_couch UNIQUE (id_couch),
    CONSTRAINT fk_couch_id FOREIGN KEY (id_couch)
        REFERENCES public.couch (id_couch) MATCH SIMPLE
        ON UPDATE NO ACTION
        ON DELETE NO ACTION
);

CREATE TABLE IF NOT EXISTS public.player
(
    id_player integer NOT NULL GENERATED BY DEFAULT AS IDENTITY ( INCREMENT 1 START 1 MINVALUE 1 MAXVALUE 2147483647 CACHE 1 ),
    name character varying(255) COLLATE pg_catalog."default",
    status integer,
    id_club integer,
    CONSTRAINT player_pkey PRIMARY KEY (id_player),
    CONSTRAINT fk_club_id FOREIGN KEY (id_club)
        REFERENCES public.club (id_club) MATCH SIMPLE
        ON UPDATE NO ACTION
        ON DELETE NO ACTION
);

CREATE TABLE IF NOT EXISTS public.club_competition
(
    id_club integer NOT NULL,
    id_competition integer NOT NULL,
    CONSTRAINT fk_competition_id FOREIGN KEY (id_competition)
        REFERENCES public.competition (id_competition) MATCH SIMPLE
        ON UPDATE NO ACTION
        ON DELETE NO ACTION,
    CONSTRAINT fk_club_id FOREIGN KEY (id_club)
        REFERENCES public.club (id_club) MATCH SIMPLE
        ON UPDATE NO ACTION
        ON DELETE NO ACTION
);

```
